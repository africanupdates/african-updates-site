CMS.registerEventListener({ name: 'preSave', handler: ({ entry }) => { const data = entry.get('data'); if (!data.get('slug')) { const title = data.get('title') || ''; const autoSlug = title .toLowerCase() .trim() .replace(/[^\w\s-]/g, '') .replace(/\s+/g, '-'); return data.set('slug', autoSlug); } return data; }, });
