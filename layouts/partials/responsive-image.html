{{- /* responsive-image.html — optimized rendering with safe fallback. Skips processing for corrupted or unresizable images, renders them directly. */ -}} {{- $r := .resource }} {{- $alt := .alt | default "" }} {{- $class := .class | default "" }} {{- $loading := .loading | default "lazy" }} {{- $sizes := .sizes | default "100vw" }} {{- $maxwidth := .maxwidth | default 1200 }} {{- if $r }} {{- $mediaMain := $r.MediaType.MainType }} {{- $mediaSubtype := $r.MediaType.SubType }} {{- /* Only attempt resize on known safe image subtypes */ -}} {{- $resizable := false }} {{- if eq $mediaMain "image" }} {{- if or (eq $mediaSubtype "jpeg") (eq $mediaSubtype "png") (eq $mediaSubtype "gif") (eq $mediaSubtype "webp") (eq $mediaSubtype "tiff") }} {{- $resizable = true }} {{- end }} {{- end }} {{- if $resizable }} {{- $ow := $r.Width }} {{- $widths := slice 400 800 1200 1600 }} {{- $srcset := slice }} {{- $fallback := "" }} {{- range $widths }} {{- if and (le . $ow) (le . $maxwidth) }} {{- $img := $r.Resize (printf "%dx webp q78" .) }} {{- $srcset = $srcset | append (printf "%s %dw" $img.RelPermalink .) }} {{- $fallback = $img.RelPermalink }} {{- end }} {{- end }} {{- if not $fallback }} {{- $img := $r.Resize (printf "%dx webp q78" $ow) }} {{- $srcset = slice (printf "%s %dw" $img.RelPermalink $ow) }} {{- $fallback = $img.RelPermalink }} {{- end }} {{- else if eq $mediaMain "image" }} {{- /* Image exists but can't be resized — render directly */ -}} {{- end }} {{- end }}
