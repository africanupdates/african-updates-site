{{- /* responsive-image.html — optimized with safe fallback for corrupt images. */ -}} {{- $r := .resource }} {{- $alt := .alt | default "" }} {{- $class := .class | default "" }} {{- $loading := .loading | default "lazy" }} {{- $sizes := .sizes | default "100vw" }} {{- $maxwidth := .maxwidth | default 1200 }} {{- if $r }} {{- if eq $r.MediaType.MainType "image" }} {{- /* Check subtype is one Hugo can process AND file is non-zero */ -}} {{- $sub := $r.MediaType.SubType }} {{- $safe := or (eq $sub "jpeg") (eq $sub "png") (eq $sub "gif") (eq $sub "webp") (eq $sub "tiff") }} {{- $hasSize := gt $r.Size 0 }} {{- if and $safe $hasSize }} {{- /* Try to get dimensions — if Width is 0 the file is corrupt, skip processing */ -}} {{- $w := $r.Width }} {{- $h := $r.Height }} {{- if and (gt $w 0) (gt $h 0) }} {{- $ow := $w }} {{- $widths := slice 400 800 1200 1600 }} {{- $srcset := slice }} {{- $fallback := "" }} {{- range $widths }} {{- if and (le . $ow) (le . $maxwidth) }} {{- $img := $r.Resize (printf "%dx webp q78" .) }} {{- $srcset = $srcset | append (printf "%s %dw" $img.RelPermalink .) }} {{- $fallback = $img.RelPermalink }} {{- end }} {{- end }} {{- if not $fallback }} {{- $img := $r.Resize (printf "%dx webp q78" $ow) }} {{- $srcset = slice (printf "%s %dw" $img.RelPermalink $ow) }} {{- $fallback = $img.RelPermalink }} {{- end }} {{- else }} {{- /* Width/Height is 0 — corrupt file, render directly without processing */ -}} {{- end }} {{- else }} {{- /* Unsupported subtype or empty file — render directly */ -}} {{- end }} {{- end }} {{- end }}
