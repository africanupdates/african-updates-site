{{- /* responsive-image.html — optimized with responsive WebP srcset. Takes a Hugo image Resource. If the source is smaller than a candidate width, that width is skipped (Hugo never upscales). Falls back to the original image URL if no processed variant is produced. Params (via dict): resource (required) — a Hugo image Resource alt, class, loading ("lazy"|"eager"), sizes, maxwidth (int, default 1200) */ -}} {{- $r := .resource }} {{- $alt := .alt | default "" }} {{- $class := .class | default "" }} {{- $loading := .loading | default "lazy" }} {{- $sizes := .sizes | default "100vw" }} {{- $maxwidth := .maxwidth | default 1200 }} {{- if $r }} {{- $ow := $r.Width }} {{- $widths := slice 400 800 1200 1600 }} {{- $srcset := slice }} {{- $fallback := $r.RelPermalink }} {{- range $widths }} {{- if and (le . $ow) (le . $maxwidth) }} {{- $img := $r.Resize (printf "%dx webp q78" .) }} {{- $srcset = $srcset | append (printf "%s %dw" $img.RelPermalink .) }} {{- $fallback = $img.RelPermalink }} {{- end }} {{- end }} {{- /* Source smaller than smallest candidate: one webp at native width */ -}} {{- if not $srcset }} {{- $img := $r.Resize (printf "%dx webp q78" $ow) }} {{- $srcset = slice (printf "%s %dw" $img.RelPermalink $ow) }} {{- $fallback = $img.RelPermalink }} {{- end }} {{- $srcsetStr := delimit $srcset ", " }} {{- end }}
