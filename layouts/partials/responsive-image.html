{{- /* responsive-image.html — optimized with responsive WebP srcset. Guards against non-image resources: if the passed resource isn't a processable image (wrong type, corrupt, octet-stream), it falls back to a plain using the resource URL instead of crashing the build. Params (via dict): resource (required) — a Hugo image Resource alt, class, loading ("lazy"|"eager"), sizes, maxwidth (int, default 1200) */ -}} {{- $r := .resource }} {{- $alt := .alt | default "" }} {{- $class := .class | default "" }} {{- $loading := .loading | default "lazy" }} {{- $sizes := .sizes | default "100vw" }} {{- $maxwidth := .maxwidth | default 1200 }} {{- if $r }} {{- $isImg := reflect.IsImageResourceProcessable $r }} {{- $ow := 0 }} {{- if $isImg }}{{ $ow = $r.Width }}{{ end }} {{- if and $isImg (gt $ow 0) }} {{- /* Genuine, processable image with real dimensions — full treatment */ -}} {{- $widths := slice 400 800 1200 1600 }} {{- $srcset := slice }} {{- $fallback := $r.RelPermalink }} {{- range $widths }} {{- if and (le . $ow) (le . $maxwidth) }} {{- $img := $r.Resize (printf "%dx webp q78" .) }} {{- $srcset = $srcset | append (printf "%s %dw" $img.RelPermalink .) }} {{- $fallback = $img.RelPermalink }} {{- end }} {{- end }} {{- if not $srcset }} {{- $img := $r.Resize (printf "%dx webp q78" $ow) }} {{- $srcset = slice (printf "%s %dw" $img.RelPermalink $ow) }} {{- $fallback = $img.RelPermalink }} {{- end }} {{- $srcsetStr := delimit $srcset ", " }} {{- else }} {{- /* Resource exists but isn't a usable image — plain img, no crash */ -}} {{- end }} {{- end }}
